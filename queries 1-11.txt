SELECT GRGR_ID,GRGR_NAME,
CSCS.CSCS_ID,CSCS.DESC,CSPI.CSPI_ID,CSPI.CSPD_CAT,
PLDS_DESC,PDDS.PDPD_ID,PDDS.PDDS-DESC
FROM CMC_GRGR_GROUP GRGR
JOIN CMC_CSCS_CLASS CSCS
ON GRGR.GRGR_CK = CSCS.GRGR_CK
JOIN CMC_CSPI_CL_PLAN CSPI 
ON CSCS.CSPI_ID = CSPI.CSPI_ID 
JOIN CMC_PLDS_PLAN_DESC PLDS
ON CSPI.CSPI_ID = PLDS.CSPI_ID
JOIN CMC_PDDS_PROD_DESC PDDS
ON GRGR.PDPD_ID = PDDS.PDPD_ID
WHERE GETDATE() BETWEEN GRGR_ORIG_EFF_DT 
AND GRGR_TERM_DT
